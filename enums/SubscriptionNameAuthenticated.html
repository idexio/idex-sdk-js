Enumeration SubscriptionNameAuthenticated

All possible WebSocket Subscription Names that require authenticated WebSocket connections.

Example

 import {
   WebSocketClient,
   SubscriptionNameAuthenticated
 } from '@idexio/idex-sdk';

 const client = new WebSocketClient({
   auth: {
     apiKey: '...',
     apiSecret: '...',
     wallet: '0x...'
   }
 })

 client.onMessage(message => {
   console.log('Received WebSocket Message: ', message)
 })

 await client.subscribeAuthenticated([
   { name: SubscriptionNameAuthenticated.positions },
   { name: SubscriptionNameAuthenticated.orders },
 ])
